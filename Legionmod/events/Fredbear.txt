		civilized = yes
		is_independant = yes
		year = 1850
		NOT = {
			has_global_flag = fredbear
		}
	}
	
	mean_time_to_happen = {
		months = 8000
		
		modifier = {
			factor = 0.65
			average_militancy = 8
		}
	}
		
		option = {
			name = "EVTOPTA19000087"
			add_country_modifier = {
				name = fredbear
				duration = 1095
			}
			any_pop = {
				limit = {
					is_accepted_culture = NO
				}
				militancy = 8
				consciousness = 8
			}
			any_neighbor_country = {
				limit = {
					NOT = { alliance_with = THIS }
				}
				add_casus_belli = {
					target = THIS
					type = cut_down_to_size
					months = 24
				}
			}
			set_global_flag = fredbear
		}
}
